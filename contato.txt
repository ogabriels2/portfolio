  <script>
    // --- Lógica de Copiar E-mail ---
    const copyBtn = document.getElementById('copyBtn');
    const emailText = "contato@ogabriels.com";

    copyBtn.addEventListener('click', () => {
      navigator.clipboard.writeText(emailText).then(() => {
        // Feedback Visual
        copyBtn.classList.add('copied');
        const icon = copyBtn.querySelector('i');
        icon.classList.replace('fa-copy', 'fa-check');
        copyBtn.style.color = 'var(--success)';

        setTimeout(() => {
          copyBtn.classList.remove('copied');
          icon.classList.replace('fa-check', 'fa-copy');
          copyBtn.style.color = '';
        }, 2000);
      });
    });

    // --- Simulação de Envio de Formulário (UX Premium) ---
    const form = document.getElementById('contactForm');
    const submitBtn = document.getElementById('submitBtn');

    form.addEventListener('submit', (e) => {
      e.preventDefault(); // Impede o envio real para demonstração

      // 1. Estado de Loading
      submitBtn.classList.add('loading');
      submitBtn.disabled = true;

      // 2. Simula tempo de rede (2 segundos)
      setTimeout(() => {
        submitBtn.classList.remove('loading');
        
        // 3. Estado de Sucesso
        submitBtn.classList.add('success');
        
        // Opcional: Resetar form
        form.reset();

        // 4. Resetar botão após 4 segundos para permitir novo envio
        setTimeout(() => {
          submitBtn.classList.remove('success');
          submitBtn.disabled = false;
        }, 4000);

      }, 2000);
    });
  </script>
